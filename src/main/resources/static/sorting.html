Quick Sort Example

Quick Sort is an efficient, divide-and-conquer sorting algorithm.


public class QuickSort {
    public static void main(String[] args) {
        int[] arr = {10, 7, 8, 9, 1, 5};
        quickSort(arr, 0, arr.length - 1);
        printArray(arr);
    }

    static void quickSort(int[] arr, int low, int high) {
        // Якщо нижній індекс менший за верхній, треба щось робити:
        if (low < high) {
            // Розподілити масив і отримати індекс опорної точки:
            int pi = partition(arr, low, high);

            // Рекурсивно сортувати підмасиви до і після опорної точки:
            quickSort(arr, low, pi - 1);
            quickSort(arr, pi + 1, high);
        }
    }

    static int partition(int[] arr, int low, int high) {
        // Взяти останній елемент як опорний:
        int pivot = arr[high];
        int i = (low - 1); // Індекс меншого елементу

        for (int j = low; j < high; j++) {
            // Якщо поточний елемент менший, ніж опорний:
            if (arr[j] < pivot) {
                i++;

                // Обміняти arr[i] та arr[j]:
                int temp = arr[i];
                arr[i] = arr[j];
                arr[j] = temp;
            }
        }

        // Обміняти arr[i+1] та arr[high] (або опорний елемент):
        int temp = arr[i + 1];
        arr[i + 1] = arr[high];
        arr[high] = temp;

        // Повернути індекс розподілу:
        return i + 1;
    }

    static void printArray(int[] arr) {
        for(int num : arr) {
            System.out.print(num + " ");
        }
        System.out.println();
    }
}
